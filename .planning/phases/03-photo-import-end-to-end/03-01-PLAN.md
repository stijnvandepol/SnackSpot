Insert this code:

```typescript
// ── Photo upload (per D-01, D-11 — after transaction, before response) ──

const photoStats: PhotoImportStats = { uploaded: 0, skipped: 0, errors: [] }

// Ensure bucket exists (Claude discretion: bucket verification)
const bucketExists = await minioClient.bucketExists(BUCKET)
if (!bucketExists) {
  await minioClient.makeBucket(BUCKET)
}

// Filter photo file entries from the ZIP (per D-12 — reuse parsed directory)
const photoEntries = directory.files.filter(
  (f: { path: string; type: string }) => f.path.startsWith('photos/') && f.type === 'File'
)

// Sequential upload loop (per D-02, matches export pattern)
for (const entry of photoEntries) {
  const storageKey = entry.path.slice('photos/'.length)
  try {
    // D-03: skip if already exists in MinIO
    const alreadyExists = await minioClient
      .statObject(BUCKET, storageKey)
      .then(() => true)
      .catch(() => false)

    if (alreadyExists) {
      photoStats.skipped++
      continue
    }

    const buffer = await entry.buffer()
    await minioClient.putObject(BUCKET, storageKey, buffer, buffer.length)
    photoStats.uploaded++
  } catch (err) {
    // D-04: failure does not abort — count and report
    // D-05: log storageKey and error reason
    const reason = err instanceof Error ? err.message : String(err)
    photoStats.errors.push(`${storageKey}: ${reason}`)
  }
}

result.photos = photoStats
```

The `result` object is typed as `ImportSummary` which now has the optional `photos` field, so `result.photos = photoStats` is valid.

IMPORTANT: The `result` is currently returned from the `db.$transaction()` callback. Since we need to mutate it after the transaction, we must ensure `result` is declared with `let` or that the assignment is on the outer variable. Looking at the code, `const result = await db.$transaction(...)` — the `result` variable itself needs to be mutable. Change `const result` to `let result` on the transaction line so `result.photos = photoStats` can be assigned. Actually, since we are adding a property to an existing object (not reassigning the variable), `const` is fine — object mutation is allowed with `const`.
